Estruturas de Dados I – Strings, Math e Date

Nesta aula, vamos explorar as estruturas de dados iniciais em JavaScript: a manipulação de strings, as funções matemáticas e o tratamento de datas com o objeto Date. Você aprenderá a buscar, cortar, juntar e transformar strings, utilizar funções do objeto Math para cálculos e arredondamentos, além de criar e manipular datas para exibir informações de forma amigável.

"Manipular dados é transformar informações em conhecimento." - Anônimo

1. Manipulação de Strings

Nesta seção, veremos como trabalhar com strings. Aprenderemos a obter o comprimento, acessar caracteres individuais e concatenar strings, utilizando tanto a forma tradicional quanto as template strings (ES6+).

Exemplo: Criação e Acesso
// Criação de strings
let titulo = "Aprender JavaScript";
let descricao = 'Estudar manipulação de strings';

console.log("Título:", titulo);
console.log("Descrição:", descricao);

// Comprimento da string
console.log("Comprimento do título:", titulo.length);

// Acesso a caracteres individuais
console.log("Primeiro caractere:", titulo[0]);
console.log("Último caractere:", titulo[titulo.length - 1]);
console.log("Caractere na posição 8:", titulo.charAt(8));
Exemplo: Concatenação e Template Strings
// Concatenação tradicional
let categoria = "Estudo";
let infoCompleta = "Categoria: " + categoria + " - " + titulo;
console.log("Concatenação tradicional:", infoCompleta);

// Template strings (ES6+)
let dataCriacao = "01/06/2025";
let resumo = `Tarefa: ${titulo} (${categoria})
Criada em: ${dataCriacao}
Descrição: ${descricao}`;
console.log("Template string:");
console.log(resumo);
Exemplo: Métodos Básicos para Busca
// Métodos para busca em strings
console.log("Posição de 'JavaScript':", titulo.indexOf("JavaScript"));
console.log("'JavaScript' está presente?", titulo.includes("JavaScript"));
console.log("Começa com 'Aprender'?", titulo.startsWith("Aprender"));
console.log("Termina com 'Script'?", titulo.endsWith("Script"));
Exemplo: Truncar Descrições

// Função que trunca uma string se ela for maior que um tamanho máximo
function truncarDescricao(texto, maxLength = 30) {
  if (texto.length <= maxLength) {
    return texto;
  }
  return texto.substring(0, maxLength) + "...";
}

let descricaoLonga = "Este é um exemplo de uma descrição muito longa que precisará ser truncada para exibição.";
console.log("Descrição truncada:", truncarDescricao(descricaoLonga));
console.log("Descrição truncada (20 caracteres):", truncarDescricao(descricaoLonga, 20));

Na parte avançada, exploramos métodos como slice, substring, split e join.

Exemplo: Slice e Substring

Exemplo: Buscar e Substituir

// Função para destacar um termo em um texto
function destacarTermo(texto, termo) {
  if (!termo) return texto;
  const regex = new RegExp(termo, 'gi');
  return texto.replace(regex, `**$&**`);
}

let busca = destacarTermo("JavaScript é uma linguagem incrível. Amo javascript!", "javascript");
console.log("Resultado com destaque:", busca);

2. Uso do Objeto Math

O objeto Math fornece constantes e métodos para operações matemáticas. Você aprenderá a utilizar constantes, métodos de arredondamento e a gerar números aleatórios.

Exemplo: Constantes e Cálculo de Área
// Exibindo constantes
console.log("Math.PI:", Math.PI);
console.log("Math.E:", Math.E);

// Cálculo da área de um círculo
const raio = 5;
const areaCirculo = Math.PI * Math.pow(raio, 2);
console.log(`Área de um círculo com raio ${raio}: ${areaCirculo}`);
Exemplo: Arredondamento

const numero = 9.7;
console.log("Math.round(9.7):", Math.round(numero));
console.log("Math.floor(9.7):", Math.floor(numero));
console.log("Math.ceil(9.7):", Math.ceil(numero));
console.log("Math.trunc(9.7):", Math.trunc(numero));

Exemplo: Números Aleatórios

// Função para gerar número aleatório entre min e max
function numeroAleatorioEntre(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}

console.log("Número aleatório entre 1 e 10:", numeroAleatorioEntre(1, 10));

3. Manipulação de Datas (Date)

O objeto Date permite criar, manipular e formatar datas. Você aprenderá a criar datas a partir de diferentes fontes (timestamp, string, componentes), extrair componentes (dia, mês, ano) e realizar cálculos, como adicionar dias e calcular a diferença entre datas.

Exemplo: Criação de Datas

// Data atual
const hoje = new Date();
console.log("Data atual:", hoje.toString());

// Data a partir de string ISO
const dataISO = new Date("2025-06-15T10:30:00");
console.log("Data a partir de string ISO:", dataISO);

// Data a partir de componentes (mês: 0 a 11)
const dataComponentes = new Date(2025, 5, 15, 10, 30, 0);
console.log("Data a partir de componentes:", dataComponentes);
Exemplo: Obtenção de Componentes e Formatação 

// Função para formatar data no formato "DD/MM/AAAA"
function formatarData(data) {
  const dia = data.getDate().toString().padStart(2, '0');
  const mes = (data.getMonth() + 1).toString().padStart(2, '0');
  const ano = data.getFullYear();
  return `${dia}/${mes}/${ano}`;
}

console.log("Data formatada:", formatarData(hoje));
Exemplo: Adicionar Dias a uma Data

// Função para adicionar dias a uma data
function adicionarDias(data, dias) {
  const novaData = new Date(data);
  novaData.setDate(data.getDate() + dias);
  return novaData;
}

const amanha = adicionarDias(hoje, 1);
console.log("Amanhã:", formatarData(amanha));
Exemplo: Diferença entre Datas

// Diferença entre duas datas (em milissegundos) e conversão para dias
const dataInicial = new Date(2025, 0, 1);
const dataFinal = new Date(2025, 11, 31);
const diferencaMs = dataFinal - dataInicial;
const diferencaDias = Math.ceil(diferencaMs / (1000 * 60 * 60 * 24));
console.log("Diferença em dias:", diferencaDias);

Ver solução

function diasAteNatal() {
  const hoje = new Date();
  let natal = new Date(hoje.getFullYear(), 11, 25);
  
  if (hoje > natal) {
    natal.setFullYear(natal.getFullYear() + 1);
  }
  
  const diferencaMs = natal - hoje;
  const diferencaDias = Math.ceil(diferencaMs / (1000 * 60 * 60 * 24));
  return diferencaDias;
}

Exercício 4: Diferença em Dias Entre Datas

Crie uma função que calcule a diferença em dias entre duas datas.

Ver solução
Resolução
function calcularDiferencaDias(data1, data2) {
  data1 = new Date(data1);
  data2 = new Date(data2);
  data1.setHours(0, 0, 0, 0);
  data2.setHours(0, 0, 0, 0);
  const diferencaMs = Math.abs(data2 - data1);
  const diferencaDias = Math.round(diferencaMs / (1000 * 60 * 60 * 24));
  return diferencaDias;
}

console.log("Diferença em dias:", calcularDiferencaDias("2025-01-01", "2025-01-15"));

Exercício 5: Formatar Data por Extenso

Crie uma função que formate uma data no padrão "Dia da semana, DD de Mês de AAAA".

Ver solução
Resolução
function formatarDataExtenso(data) {
  if (typeof data === 'string') {
    data = new Date(data);
  }
  
  const diasSemana = [
    'Domingo', 'Segunda-feira', 'Terça-feira', 'Quarta-feira',
    'Quinta-feira', 'Sexta-feira', 'Sábado'
  ];
  const meses = [
    'Janeiro', 'Fevereiro', 'Março', 'Abril', 'Maio', 'Junho',

  const dia = data.getDate();
  const mes = meses[data.getMonth()];
  const ano = data.getFullYear();

}

Ver solução
Resolução
function capitalizarPalavras(texto) {
  const palavras = texto.split(" ");
  for (let i = 0; i < palavras.length; i++) {
    if (palavras[i].length > 0) {
      palavras[i] = palavras[i][0].toUpperCase() + palavras[i].slice(1);
    }
  }
  return palavras.join(" ");
}

console.log("Texto capitalizado:", capitalizarPalavras("olá mundo, como vai você?"));

Exercício 8: Extrair Hashtags

Crie uma função que extraia todas as hashtags de um texto (retorne apenas as palavras sem o símbolo #).

Ver solução
Resolução
function extrairHashtags(texto) {
  const regex = /#(\w+)/g;
  const hashtags = [];
  let match;
  while ((match = regex.exec(texto)) !== null) {
    hashtags.push(match[1]);
  }
  return hashtags;
}

console.log("Hashtags extraídas:", extrairHashtags("Estudar #javascript e #react para o projeto #taskmaster"));

Exercício 9: Encontrar Maior e Menor Número

Crie uma função que receba um array de números e retorne um objeto com o maior e o menor número, sem utilizar o operador spread.

Ver solução
Resolução
function encontrarMaiorEMenor(numeros) {
  if (numeros.length === 0) return { maior: null, menor: null };
  let maior = numeros[0];
  let menor = numeros[0];
  for (let i = 1; i < numeros.length; i++) {
    if (numeros[i] > maior) maior = numeros[i];
    if (numeros[i] < menor) menor = numeros[i];
  }
  return { maior, menor };
}

console.log("Maior e menor:", encontrarMaiorEMenor([15, 7, 23, 42, 8, 19, 3]));
